218  // @brief multiplies matrices based on a dot-product computation with a rectangular result matrix
219  template <typename Derived1, typename Derived2>
220  void _multiplyNM_dotProduct(
221  Eigen::PlainObjectBase<Derived1> &leftMatrix,
222  Eigen::PlainObjectBase<Derived2> &rightMatrix,
223  Eigen::PlainObjectBase<Derived2> &result,
224  const std::vector<int> & offsets,
225  int p, int q, int r)
226  {
227  PRECICE_TRACE();
228  for (int i = 0; i < leftMatrix.rows(); i++) {
229  int rank = 0;
230  // find rank of processor that stores the result
231  // the second while is necessary if processors with no vertices are present
232  // Note: the >'=' here is crucial: In case some procs do not have any vertices,
233  // this while loop continues incrementing rank if entries in offsets are equal, i.e.,
234  // it runs to the next non-empty proc.
235  while (i >= offsets[rank + 1])
236  rank++;
237 
238  Eigen::VectorXd lMRow = leftMatrix.row(i);
239 
240  for (int j = 0; j < r; j++) {
241 
242  Eigen::VectorXd rMCol = rightMatrix.col(j);
243  double res_ij = utils::MasterSlave::dot(lMRow, rMCol);
244 
245  // find proc that needs to store the result.
246  int local_row;
247  if (utils::MasterSlave::getRank() == rank) {
248  local_row = i - offsets[rank];
249  result(local_row, j) = res_ij;
250  }
251  }
252  }

257  void _multiplyNM_block(
258  Eigen::PlainObjectBase<Derived1> &leftMatrix,
259  Eigen::PlainObjectBase<Derived2> &rightMatrix,

261  const std::vector<int> & offsets,
262  int p, int q, int r)
263  {
264  PRECICE_TRACE();
265 
266  // ensure that both matrices are stored in the same order. Important for reduce function, that adds serialized data.
267  PRECICE_ASSERT(static_cast<int>(leftMatrix.IsRowMajor) == static_cast<int>(rightMatrix.IsRowMajor),
268  leftMatrix.IsRowMajor, rightMatrix.IsRowMajor);
269 
270  // multiply local block (saxpy-based approach)
271  // dimension: (n_global x n_local) * (n_local x m) = (n_global x m)
272  Eigen::MatrixXd block = Eigen::MatrixXd::Zero(p, r);
273  block.noalias() = leftMatrix * rightMatrix;
274 
275  // all blocks have size (n_global x m)
276  // Note: if procs have no vertices, the block size remains (n_global x m), however,
277  // it must be initialized with zeros, so zeros are added for those procs)
278 
279  // sum up blocks in master, reduce
280  Eigen::MatrixXd summarizedBlocks = Eigen::MatrixXd::Zero(p, r);
281  utils::MasterSlave::reduceSum(block.data(), summarizedBlocks.data(), block.size());
282 
283  // slaves wait to receive their local result
284  if (utils::MasterSlave::isSlave()) {
285  if (result.size() > 0)
286  utils::MasterSlave::_communication->receive(result.data(), result.size(), 0);
287  }
288 
289  // master distributes the sub blocks of the results
290  if (utils::MasterSlave::isMaster()) {
291  // distribute blocks of summarizedBlocks (result of multiplication) to corresponding slaves
292  result = summarizedBlocks.block(0, 0, offsets[1], r);
293 
294  for (int rankSlave = 1; rankSlave < utils::MasterSlave::getSize(); rankSlave++) {
295  int off = offsets[rankSlave];
296  int send_rows = offsets[rankSlave + 1] - offsets[rankSlave];
297 
298  if (summarizedBlocks.block(off, 0, send_rows, r).size() > 0) {
299  // necessary to save the matrix-block that is to be sent in a temporary matrix-object
300  // otherwise, the send routine walks over the bounds of the block (matrix structure is still from the entire matrix)
301  Eigen::MatrixXd sendBlock = summarizedBlocks.block(off, 0, send_rows, r);
302  utils::MasterSlave::_communication->send(sendBlock.data(), sendBlock.size(), rankSlave);
303  }
304  }
305  }
