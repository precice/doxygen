96 bool FindClosest::determineClosest()
97 {
98  PRECICE_TRACE(_searchpoint);
99  using math::greater;
100  _closest = ClosestElement(_searchpoint.size());
101  _closest.distance = std::numeric_limits<double>::max();
102  int closestType = -1;
103  if (greater(_closest.distance, _findClosestVertex.getEuclidianDistance())) {
104  _closest.distance = _findClosestVertex.getEuclidianDistance();
105  closestType = 0;
106  }
107  if (greater(_closest.distance, _findClosestEdge.getEuclidianDistance())) {
108  _closest.distance = _findClosestEdge.getEuclidianDistance();
109  closestType = 1;
110  }
111  if (greater(_closest.distance, _findClosestTriangle.getEuclidianDistance())) {
112  _closest.distance = _findClosestTriangle.getEuclidianDistance();
113  closestType = 2;
114  }
115  // Assign all properties to _closest
116  Eigen::VectorXd normal = Eigen::VectorXd::Zero(_searchpoint.size());
117  if (closestType == 0) { // Vertex
118  mesh::Vertex &vertex = _findClosestVertex.getClosestVertex();
119  _closest.vectorToElement = vertex.getCoords() - _searchpoint;
120  normal = vertex.getNormal();
121  InterpolationElement element;
122  element.element = &vertex;
123  element.weight = 1.0;
124  _closest.interpolationElements.push_back(element);
125  } else if (closestType == 1) { // Edge
126  mesh::Edge &edge = _findClosestEdge.getClosestEdge();
127  _closest.vectorToElement = _findClosestEdge.getVectorToProjectionPoint();
128  normal = edge.getNormal();
129  InterpolationElement element0, element1;
130  element0.element = &edge.vertex(0);
131  element1.element = &edge.vertex(1);
132  element0.weight = _findClosestEdge.getProjectionPointParameter(0);
133  element1.weight = _findClosestEdge.getProjectionPointParameter(1);
134  _closest.interpolationElements.push_back(element0);
135  _closest.interpolationElements.push_back(element1);
136  } else if (closestType == 2) { // Triangle
137  mesh::Triangle &triangle = _findClosestTriangle.getClosestTriangle();
138  _closest.vectorToElement = _findClosestTriangle.getVectorToProjectionPoint();
139  normal = triangle.getNormal();
140  InterpolationElement element0, element1, element2;
141  element0.element = &triangle.vertex(0);
142  element1.element = &triangle.vertex(1);
143  element2.element = &triangle.vertex(2);
144  element0.weight = _findClosestTriangle.getProjectionPointParameter(0);
145  element1.weight = _findClosestTriangle.getProjectionPointParameter(1);
146  element2.weight = _findClosestTriangle.getProjectionPointParameter(2);
147  _closest.interpolationElements.push_back(element0);
148  _closest.interpolationElements.push_back(element1);
149  _closest.interpolationElements.push_back(element2);
150  } else {
151  return false;
152  }
153  // Flip sign of distance, depending on normal of closest element
154  if (_closest.vectorToElement.dot(normal) > 0.0) {
155  _closest.distance *= -1.0;
156  }
157  return true;
158 }
