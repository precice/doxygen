124 void TestContext::initializeMPI(const TestContext::Participants &participants)
125 {
126  auto baseComm = Par::current();
127  const int globalRank = baseComm->rank();
128  const int available = baseComm->size();
129  const int required = std::accumulate(participants.begin(), participants.end(), 0, [](int total, const Participant &next) { return total + next.size; });
130  if (required > available) {
131  throw std::runtime_error{"This test requests " + std::to_string(required) + " ranks, but there are only " + std::to_string(available) + " available"};
132  }
133 
134  // Restrict the communicator to the total required size
135  Par::restrictCommunicator(required);
136 
137  // Mark all unnecessary ranks as invalid and return
138  if (globalRank >= required) {
139  invalid = true;
140  return;
141  }
142 
143  // If there was only a single participant requested, then update its info and we are done.
144  if (participants.size() == 1) {
145  auto &participant = participants.front();
146  if (!invalid) {
147  setContextFrom(participant, globalRank);
148  }
149  return;
150  }
151 
152  // If there were multiple participants requested, we need to split the restricted comm
153  if (participants.size() > 1) {
154  int offset = 0;
155  for (const auto &participant : participants) {
156  const auto localRank = globalRank - offset;
157  // Check if my global rank maps to this participant
158  if (localRank < participant.size) {
159  Par::splitCommunicator(participant.name);
160  setContextFrom(participant, localRank);
161  return;
162  }
163  offset += participant.size;
164  }
165  }
166 }
