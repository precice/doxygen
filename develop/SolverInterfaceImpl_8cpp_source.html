854 void SolverInterfaceImpl::setMeshQuad(
855  int meshID,
856  int firstEdgeID,
857  int secondEdgeID,
858  int thirdEdgeID,
859  int fourthEdgeID)
860 {
861  PRECICE_TRACE(meshID, firstEdgeID, secondEdgeID, thirdEdgeID,
862  fourthEdgeID);
863  PRECICE_CHECK(_dimensions == 3, "setMeshQuad is only possible for 3D cases."
864  " Please set the dimension to 3 in the preCICE configuration file.");
865  PRECICE_REQUIRE_MESH_MODIFY(meshID);
866  MeshContext &context = _accessor->meshContext(meshID);
867  if (context.meshRequirement == mapping::Mapping::MeshRequirement::FULL) {
868  mesh::PtrMesh &mesh = context.mesh;
869  using impl::errorInvalidEdgeID;
870  PRECICE_CHECK(mesh->isValidEdgeID(firstEdgeID), errorInvalidEdgeID(firstEdgeID));
871  PRECICE_CHECK(mesh->isValidEdgeID(secondEdgeID), errorInvalidEdgeID(secondEdgeID));
872  PRECICE_CHECK(mesh->isValidEdgeID(thirdEdgeID), errorInvalidEdgeID(thirdEdgeID));
873  PRECICE_CHECK(mesh->isValidEdgeID(fourthEdgeID), errorInvalidEdgeID(fourthEdgeID));
874 
875  PRECICE_CHECK(utils::unique_elements(utils::make_array(firstEdgeID, secondEdgeID, thirdEdgeID, fourthEdgeID)),
876  "The four edge ID's are not unique. Please check that the edges that form the quad are correct.");
877 
878  auto chain = mesh::asChain(utils::make_array(
879  &mesh->edges()[firstEdgeID], &mesh->edges()[secondEdgeID],
880  &mesh->edges()[thirdEdgeID], &mesh->edges()[fourthEdgeID]));
881  PRECICE_CHECK(chain.connected, "The four edges are not connect. Please check that the edges that form the quad are correct.");
882 
883  auto coords = mesh::coordsFor(chain.vertices);
884  PRECICE_CHECK(utils::unique_elements(coords),
885  "The four vertices that form the quad are not unique. The resulting shape may be a point, line or triangle."
886  "Please check that the adapter sends the four unique vertices that form the quad, or that the mesh on the interface "
887  "is composed of planar quads.");
888 
889  auto convexity = math::geometry::isConvexQuad(coords);
890  PRECICE_CHECK(convexity.convex, "The given quad is not convex. "
891  "Please check that the adapter send the four correct vertices or that the interface is composed of planar quads.");
892 
893  // Use the shortest diagonal to split the quad into 2 triangles.
894  // The diagonal to be used with edges (1, 2) and (0, 3) of the chain
895  double distance1 = (coords[0] - coords[2]).norm();
896  // The diagonal to be used with edges (0, 1) and (2, 3) of the chain
897  double distance2 = (coords[1] - coords[3]).norm();
898 
899  // The new edge, e[4], is the shortest diagonal of the quad
900  if (distance1 <= distance2) {
901  auto &diag = mesh->createUniqueEdge(*chain.vertices[0], *chain.vertices[2]);
902  mesh->createTriangle(*chain.edges[3], *chain.edges[0], diag);
903  mesh->createTriangle(*chain.edges[1], *chain.edges[2], diag);
904  } else {
905  auto &diag = mesh->createUniqueEdge(*chain.vertices[1], *chain.vertices[3]);
906  mesh->createTriangle(*chain.edges[0], *chain.edges[1], diag);
907  mesh->createTriangle(*chain.edges[2], *chain.edges[3], diag);
908  }
909  }
910 }

932  PRECICE_CHECK(mesh.isValidVertexID(fourthVertexID), errorInvalidVertexID(fourthVertexID));
933 
934  auto vertexIDs = utils::make_array(firstVertexID, secondVertexID, thirdVertexID, fourthVertexID);
935  PRECICE_CHECK(utils::unique_elements(vertexIDs), "The four vertex ID's are not unique. Please check that the vertices that form the quad are correct.");
936

950  // The order now identifies all outer edges of the quad.

952  auto &edge1 = mesh.createUniqueEdge(*reordered[1], *reordered[2]);
953  auto &edge2 = mesh.createUniqueEdge(*reordered[2], *reordered[3]);

1399 void SolverInterfaceImpl::computePartitions()
1400 {

1406 
1407  std::sort(contexts.begin(), contexts.end(),
1408  [](MeshContext const *const lhs, MeshContext const *const rhs) -> bool {
1409  return lhs->mesh->getName() < rhs->mesh->getName();
1410  });
1411 
1412  for (MeshContext *meshContext : contexts) {
1413  meshContext->partition->communicate();
1414  }
1415 
1416  // for two-level initialization, there is also still communication in partition::compute()

1431  return meshContext->provideMesh;

1437  meshContext->mesh->computeState();
