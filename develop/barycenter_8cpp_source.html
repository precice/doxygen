23  const int dimensions = edgeA.size();
24  PRECICE_ASSERT(dimensions == edgeB.size() && dimensions == edgeNormal.size() && dimensions == location.size(),
25  "The inputs need to have the same dimensions.");
26  PRECICE_ASSERT((dimensions == 2) || (dimensions == 3), dimensions);
27 
28  Vector2d barycentricCoords;
29  VectorXd projected = VectorXd::Zero(dimensions);
30  Vector2d a, b, ab, c, d;
31  bool collinear = false;
32 
33  if (dimensions == 2) {
34  // Get parameters for parametric edge representation: p(s) = a + s(b-a)
35  a = edgeA.head<2>();
36  b = edgeB.head<2>();
37  ab = b;
38  ab -= a;
39  // Same for intersecting normal from searchpoint: q(t) = c + t(d - c)
40  c = location.head<2>();
41  d = location.head<2>();
42  d += edgeNormal;
43  collinear = math::geometry::collinear(a, b, c);
44  if (collinear) {
45  // From p(s) = a + s(b-a) we get: s = (p(s) - a) / (b-a)
46  int iMax;
47  ab.cwiseAbs().maxCoeff(&iMax);
48  PRECICE_ASSERT(!math::equals(ab(iMax), 0.0));
49  barycentricCoords[0] = (location(iMax) - a(iMax)) / ab(iMax);
50  barycentricCoords[1] = 1.0 - barycentricCoords[0];
51  projected = location.head<2>();
52  std::swap(barycentricCoords(0), barycentricCoords(1));
53  return {barycentricCoords, projected};
54  }
55  } else { // 3D
56  PRECICE_ASSERT(dimensions == 3, dimensions);
57  // Get parameters for parametric triangle representation: p(s) = a + s(b-a)
58  Vector3d a3D = edgeA;
59  Vector3d b3D = edgeB;
60  Vector3d c3D = location;
61  Vector3d ab3D = b3D - a3D;
62  Vector3d ac3D = c3D - a3D;
63  collinear = math::geometry::collinear(a3D, b3D, c3D);
64  if (collinear) {
65  // From p(s) = a + s(b-a) we get: s = (p(s) - a) / (b-a)
66  int iMax;
67  ab3D.cwiseAbs().maxCoeff(&iMax);
68  PRECICE_ASSERT(!math::equals(ab3D[iMax], 0.0));
69  barycentricCoords[0] =
70  (location(iMax) - edgeA(iMax)) / ab3D(iMax);
71  barycentricCoords[1] = 1.0 - barycentricCoords[0];
72  projected = location;
73  std::swap(barycentricCoords(0), barycentricCoords(1));
74  return {barycentricCoords, projected};
75  }
76  // Project parameters to 2D, where the projection plane is determined from
77  // the normal direction, in order to prevent "faulty" projections.
78  Vector3d normal;
79  normal = ab3D.cross(ac3D);
80  int indexToRemove;
81  normal.cwiseAbs().maxCoeff(&indexToRemove);
82  int indices[2];
83  if (indexToRemove == 0) {
84  indices[0] = 1;
85  indices[1] = 2;
86  } else if (indexToRemove == 1) {
87  indices[0] = 0;
88  indices[1] = 2;
89  } else {
90  PRECICE_ASSERT(indexToRemove == 2, indexToRemove);
91  indices[0] = 0;
92  indices[1] = 1;
93  }
94  a << a3D[indices[0]], a3D[indices[1]];
95  b << b3D[indices[0]], b3D[indices[1]];
96  ab << ab3D[indices[0]], ab3D[indices[1]];
97  c << c3D[indices[0]], c3D[indices[1]];
98  // 3D normal might be projected out, hence, compute new 2D edge normal
99  Vector2d normal2D(-1.0 * ab(1), ab(0));
100  d = c + normal2D;
101  }
102 
103  // Compute denominator for solving 2x2 equation system
104  double D = a(0) * (d(1) - c(1)) + b(0) * (c(1) - d(1)) + d(0) * ab(1) - c(0) * ab(1);
105  PRECICE_ASSERT(not math::equals(D, 0.0), a, b, c, d, ab); // D == 0 would imply "normal // edge"
106 
107  // Compute triangle segment parameter s, which is in [0, 1], if the
108  // intersection point is within the triangle.
109  barycentricCoords[0] = (a(0) * (d(1) - c(1)) +
110  c(0) * (a(1) - d(1)) +
111  d(0) * (c(1) - a(1))) /
112  D;
113  barycentricCoords[1] = 1.0 - barycentricCoords[0];
114 
115  // Compute coordinates of projected point, which has to be done dimension
116  // dependent again:
117  if (dimensions == 2) {
118  projected = ab; // = b - a
119  projected *= barycentricCoords[0]; // = bary0 * (b - a)
120  projected += a; // = a + bary0 * (b - a)
121  } else {
122  projected = edgeB; // = b
123  projected -= edgeA; // = b - a
124  projected *= barycentricCoords[0]; // = bary0 * (b - a)
125  projected += edgeA; // = a + bary0 * (b - a)
126  }
127 
128  std::swap(barycentricCoords(0), barycentricCoords(1));
129  return {barycentricCoords, projected};
130 }

132 BarycentricCoordsAndProjected calcBarycentricCoordsForTriangle(
133  const Eigen::VectorXd &a,
134  const Eigen::VectorXd &b,
135  const Eigen::VectorXd &c,
136  const Eigen::VectorXd &normal,
137  const Eigen::VectorXd &location)
138 {
139  using Eigen::Vector2d;
140  using Eigen::Vector3d;
141 
142  // Parametric representation for triangle plane:
143  // (x, y, z) * normal = d
144  const double d = normal.dot(a);
145 
146  // Parametric description of line from searchpoint orthogonal to triangle:
147  // location + t * normal = x (where t is parameter)
148  // Determine t such that x lies on triangle plane:
149  const double t = d - location.dot(normal) / normal.dot(normal);
150 
151  // Compute projected point with parameter t:
152  Vector3d projected = normal;
153  projected *= t;
154  projected += location;
155 
156  // Project everything to 2D
157  int iMax;
158  normal.cwiseAbs().maxCoeff(&iMax);
159  int indices[2];
160  if (iMax == 0) {
161  indices[0] = 1;
162  indices[1] = 2;
163  } else if (iMax == 1) {
164  indices[0] = 0;
165  indices[1] = 2;
166  } else {
167  PRECICE_ASSERT(iMax == 2, iMax);
168  indices[0] = 0;
169  indices[1] = 1;
170  }
171  Vector2d a2D(a[indices[0]],
172  a[indices[1]]);
173  Vector2d b2D(b[indices[0]],
174  b[indices[1]]);
175  Vector2d c2D(c[indices[0]],
176  c[indices[1]]);
177  Vector2d projected2D(projected[indices[0]], projected[indices[1]]);
178  // Compute barycentric coordinates by solving linear 3x3 system
179  Vector3d rhs(projected2D(0), projected2D(1), 1);
180  Eigen::Matrix<double, 3, 3> A;
181  A << a2D(0), b2D(0), c2D(0),
182  a2D(1), b2D(1), c2D(1),
183  1, 1, 1;
184  Vector3d barycentricCoords = A.colPivHouseholderQr().solve(rhs);
185 
186  return {barycentricCoords, projected};
187 }
