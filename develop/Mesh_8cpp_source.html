187  for (PtrData data : _data) {

214 void Mesh::computeState()
215 {
216  PRECICE_TRACE(_name);
217  PRECICE_ASSERT(_dimensions == 2 || _dimensions == 3, _dimensions);
218 
219  // Compute normals only if faces to derive normal information are available
220  size_t size2DFaces = _edges.size();
221  size_t size3DFaces = _triangles.size();
222  if (_dimensions == 2 && size2DFaces == 0) {
223  return;
224  }
225  if (_dimensions == 3 && size3DFaces == 0) {
226  return;
227  }
228 
229  // Compute (in 2D) edge normals
230  if (_dimensions == 2) {
231  for (Edge &edge : _edges) {
232  Eigen::VectorXd weightednormal = edge.computeNormal(_flipNormals);
233 
234  // Accumulate normal in associated vertices
235  for (int i = 0; i < 2; i++) {
236  Eigen::VectorXd vertexNormal = edge.vertex(i).getNormal();
237  vertexNormal += weightednormal;
238  edge.vertex(i).setNormal(vertexNormal);
239  }
240  }
241  }
242 
243  if (_dimensions == 3) {
244  // Compute normals
245  for (Triangle &triangle : _triangles) {
246  PRECICE_ASSERT(triangle.vertex(0) != triangle.vertex(1),
247  triangle.vertex(0), triangle.getID());
248  PRECICE_ASSERT(triangle.vertex(1) != triangle.vertex(2),
249  triangle.vertex(1), triangle.getID());
250  PRECICE_ASSERT(triangle.vertex(2) != triangle.vertex(0),
251  triangle.vertex(2), triangle.getID());
252 
253  // Compute normals
254  Eigen::VectorXd weightednormal = triangle.computeNormal(_flipNormals);
255 
256  // Accumulate area-weighted normal in associated vertices and edges
257  for (int i = 0; i < 3; i++) {
258  triangle.edge(i).setNormal(triangle.edge(i).getNormal() + weightednormal);
259  triangle.vertex(i).setNormal(triangle.vertex(i).getNormal() + weightednormal);
260  }
261  }
262 
263  // Normalize edge normals (only done in 3D)
264  for (Edge &edge : _edges) {
265  // there can be cases when an edge has no adjacent triangle though triangles exist in general (e.g. after filtering)
266  edge.setNormal(edge.getNormal().normalized());
267  }
268  }
269 
270  for (Vertex &vertex : _vertices) {
271  // there can be cases when a vertex has no edge though edges exist in general (e.g. after filtering)
272  vertex.setNormal(vertex.getNormal().normalized());
273  }
274 }

355 void Mesh::addMesh(
356  Mesh &deltaMesh)
357 {
358  PRECICE_TRACE();
359  PRECICE_ASSERT(_dimensions == deltaMesh.getDimensions());
360 
361  boost::container::flat_map<int, Vertex *> vertexMap;
362  vertexMap.reserve(deltaMesh.vertices().size());
363  Eigen::VectorXd coords(_dimensions);
364  for (const Vertex &vertex : deltaMesh.vertices()) {
365  coords = vertex.getCoords();
366  Vertex &v = createVertex(coords);
367  v.setGlobalIndex(vertex.getGlobalIndex());
368  if (vertex.isTagged())
369  v.tag();
370  v.setOwner(vertex.isOwner());
371  PRECICE_ASSERT(vertex.getID() >= 0, vertex.getID());
372  vertexMap[vertex.getID()] = &v;
373  }
374 
375  boost::container::flat_map<int, Edge *> edgeMap;
376  edgeMap.reserve(deltaMesh.edges().size());
377  // you cannot just take the vertices from the edge and add them,
378  // since you need the vertices from the new mesh
379  // (which may differ in IDs)
380  for (const Edge &edge : deltaMesh.edges()) {
381  int vertexIndex1 = edge.vertex(0).getID();
382  int vertexIndex2 = edge.vertex(1).getID();
383  PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
384  (vertexMap.count(vertexIndex2) == 1));
385  Edge &e = createEdge(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2]);
386  edgeMap[edge.getID()] = &e;
387  }
388 
389  if (_dimensions == 3) {
390  for (const Triangle &triangle : deltaMesh.triangles()) {
391  int edgeIndex1 = triangle.edge(0).getID();
392  int edgeIndex2 = triangle.edge(1).getID();
393  int edgeIndex3 = triangle.edge(2).getID();
394  PRECICE_ASSERT((edgeMap.count(edgeIndex1) == 1) &&
395  (edgeMap.count(edgeIndex2) == 1) &&
396  (edgeMap.count(edgeIndex3) == 1));
397  createTriangle(*edgeMap[edgeIndex1], *edgeMap[edgeIndex2], *edgeMap[edgeIndex3]);
398  }
399  }
400  meshChanged(*this);
401 }
