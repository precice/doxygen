325 
326  for (const InterpolationElements &elems : _weights) {

328  if (!math::equals(elem.weight, 0.0)) {
329  tagged.insert(elem.element);
330  }

332  // Shortcut if all vertices are tagged
333  if (tagged.size() == max_count) {

335  }

339  for (auto &v : origins->vertices()) {
340  if (tagged.count(&v) == 1) {
341  v.tag();
342  }
343  }
